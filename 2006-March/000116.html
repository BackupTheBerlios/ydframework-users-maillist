don't worry, it's really easy =)
i normally do something like this:

function actionWhatever() {
    //get some data
    $dbo = YDDatabaseObject::getInstance('whatever');
    $dbo->find();
    $results = $dbo->getResults();
    
    //now create the form and set the form values
    $form = YDForm ('whatever');
    $form->setDefaults ($results);

    $this->template->assignForm( 'whatever', $form ); 
    $this->template->display ( 'whatever.tpl' ); 

}
